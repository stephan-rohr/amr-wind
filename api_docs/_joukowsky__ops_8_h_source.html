108 struct ComputeForceOp<Joukowsky, ActSrcDisk>
109 {
110  void operator()(Joukowsky::DataType& data)
111  {
112  // Equation comments refer to Sorenson 2020 (full reference above)
113  auto& grid = data.grid();
114  auto& ddata = data.meta();
115  const amrex::Real machine_eps = vs::DTraits<amrex::Real>::eps();
116  ddata.disk_force *= 0.0;
117 
118  const amrex::Real uInfSqr = base::compute_reference_velocity_sqr(ddata);
119  const amrex::Real U_ref = std::max(std::sqrt(uInfSqr), machine_eps);
120  base::set_thrust_coefficient(ddata, uInfSqr);
121  joukowsky::set_current_angular_velocity(ddata, uInfSqr);
122 
123  const amrex::Real Ct = ddata.current_ct;
124  const amrex::Real dx = ddata.dr / ddata.radius();
125 
126  const ops::base::AreaComputer area(
127  ddata.radius(), ddata.num_vel_pts_r, ddata.num_vel_pts_t);
128 
129  const amrex::Real lambda =
130  ddata.radius() * ddata.current_angular_velocity / U_ref;
131 
132  ddata.current_tip_speed_ratio = lambda;
133 
134  const amrex::Real lambda_2 = lambda * lambda;
135 
136  // step 0: compute tip correction based on current wind speed (eq 9)
137  if (ddata.use_tip_correction) {
138  auto& f = ddata.tip_correction;
139  for (int i = 0; i < f.size(); ++i) {
140  const auto x = (i + 0.5) * dx;
141  f[i] = 2.0 / M_PI *
142  std::acos(std::exp(
143  -0.5 * ddata.num_blades * std::sqrt(1.0 + lambda_2) *
144  (1.0 - x)));
145  }
146  }
147 
148  // step 1: compute Ct, a1 and a2 coefficients (eq 16)
149  amrex::Real a1 = 0.0;
150  amrex::Real a2 = 0.0;
151 
152  for (int ip = 0; ip < ddata.num_vel_pts_r; ip++) {
153  const amrex::Real x = std::max((ip + 0.5) * dx, machine_eps);
154 
155  a1 += std::pow(ddata.tip_correction[ip], 2.0) *
156  std::pow(ddata.root_correction[ip], 2.0) / x * dx;
157 
158  a2 += ddata.tip_correction[ip] * ddata.root_correction[ip] * x * dx;
159  }
160 
161  // step 2: determine the circulation (q0) from a1, a2 and Ct (eq 17)
162 
163  const amrex::Real a2_2 = a2 * a2;
164 
165  const amrex::Real term1 = 16.0 * lambda_2 * a2_2;
166  const amrex::Real term2 = 8.0 * Ct * a1;
167  const amrex::Real term3 = 4.0 * lambda * a2;
168 
169  const amrex::Real numerator = std::sqrt(term1 + term2) - term3;
170 
171  const amrex::Real denominator = std::max(4.0 * a1, machine_eps);
172 
173  const amrex::Real q0 = numerator / denominator;
174 
175  // step 3: compute normal force (fz) and azimuthal force (f_theta) (eq
176  // 13) and cp (eq 20)
177 
178  VecSlice disk_velocity = ::amr_wind::utils::slice(
179  grid.vel, ddata.num_force_pts, ddata.num_force_pts);
180 
181  amrex::Real moment = 0.0;
182 
183  int ip = 0;
184 
185  for (int i = 0; i < ddata.num_vel_pts_r; i++) {
186  const amrex::Real& F = ddata.tip_correction[i];
187  const amrex::Real& g = ddata.root_correction[i];
188 
189  const amrex::Real x = std::max((i + 0.5) * dx, machine_eps);
190  // const amrex::Real& g = x;
191 
192  const amrex::Real f_z =
193  q0 * g * F / x * (lambda * x + 0.5 * q0 * g * F / x);
194 
195  ddata.f_normal[i] = f_z;
196 
197  const auto dArea = area.area_section(i);
198 
199  for (int j = 0; j < ddata.num_vel_pts_t; j++, ip++) {
200  const auto point_current = grid.pos[ip];
201 
202  // TODO add sign convention for rotation (+/- RPM)
203  const auto theta_vec = utils::compute_tangential_vector(
204  ddata.center, ddata.normal_vec, point_current);
205 
206  // normal vec by definition is opposite of the wind direction
207  // so we need to flip the sign to give the actual disk velocity
208  const amrex::Real u_disk_ij =
209  -ddata.normal_vec & disk_velocity[ip];
210 
211  const amrex::Real f_theta = u_disk_ij / U_ref * q0 * g * F / x;
212  ddata.f_theta[ip] = f_theta;
213 
214  // eq 18
215  moment += x * ddata.radius() * f_theta * dArea;
216 
217  grid.force[ip] =
218  (f_z * ddata.normal_vec + f_theta * theta_vec) *
219  ddata.density * uInfSqr * dArea;
220 
221  ddata.disk_force = ddata.disk_force + grid.force[ip];
222  }
223  }
224 
225  // equation 20
226  const amrex::Real eq_20_denominator = std::max(
227  0.5 * M_PI * ddata.density * std::pow(ddata.radius(), 2) *
228  std::pow(U_ref, 3),
229  machine_eps);
230 
231  ddata.current_cp = ddata.density * uInfSqr * moment *
232  ddata.current_angular_velocity / eq_20_denominator;
233  }
234 };
