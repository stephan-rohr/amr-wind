137 template <typename ActTrait>
138 struct UpdateVelOp<
139  ActTrait,
140  ActSrcDisk,
141  std::enable_if_t<std::is_base_of<DiskType, ActTrait>::value>>
142 {
143  void operator()(typename ActTrait::DataType& data)
144  {
145  auto& meta = data.meta();
146  const auto& grid = data.grid();
147  meta.reference_velocity = {0.0, 0.0, 0.0};
148  meta.mean_disk_velocity = {0.0, 0.0, 0.0};
149  meta.density = 0.0;
150  auto& refVel = meta.reference_velocity;
151  auto& diskVel = meta.mean_disk_velocity;
152  auto& density = meta.density;
153  const int np = meta.num_vel_pts / 2;
154  const int num_r = meta.num_vel_pts_r;
155  const int num_t = meta.num_vel_pts_t;
156 
157  const base::AreaComputer area(meta.radius(), num_r, num_t);
158 
159  for (int i = 0; i < num_r; i++) {
160  const auto weight = area.weight(i);
161  for (int j = 0; j < num_t; ++j) {
162  density += grid.density[j + i * num_t] * weight;
163  refVel = refVel + grid.vel[j + i * num_t] * weight;
164  diskVel = diskVel + grid.vel[j + i * num_t + np] * weight;
165  }
166  }
167  }
168 };
