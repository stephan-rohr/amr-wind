30 struct ComputeForceOp<UniformCt, ActSrcDisk>
31 {
32  void operator()(UniformCt::DataType& data)
33  {
34  auto& grid = data.grid();
35  auto& ddata = data.meta();
36  const auto& normal = ddata.normal_vec;
37  const amrex::Real uInfSqr = base::compute_reference_velocity_sqr(ddata);
38  base::set_thrust_coefficient(ddata, uInfSqr);
39 
40  const int npts = ddata.num_force_pts;
41  const amrex::Real rho = ddata.density;
42  const amrex::Real aeroPressure = 0.5 * uInfSqr * rho * ddata.current_ct;
43  const amrex::Real dr = ddata.dr;
44 
45  for (int ip = 0; ip < npts; ++ip) {
46  const amrex::Real r = (ip + 0.5) * dr;
47  const amrex::Real rp = r + dr * 0.5;
48  const amrex::Real rm = r - dr * 0.5;
49  const amrex::Real a = ::amr_wind::utils::pi() * (rp * rp - rm * rm);
50  // disk force should always point in direction of the normal
51  grid.force[ip] = (aeroPressure * a) * normal;
52  }
53  }
54 };
